  hash := make(map[int]int)
  for x := range numbers {
    tt := target - nums[x]
    if i, ok := hash[tt]; ok {

func twoSum(numbers []int, target int) []int {
    left, right := 0, len(numbers)-1
    for left < right {
        sum := numbers[left] + numbers[right]
        if sum == target {
            return []int{left+1, right+1}
        } else if sum > target {
            right--
        } else {
            left++
        }
    }
    return []int{}
}
